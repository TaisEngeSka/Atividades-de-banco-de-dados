create database "AtividadeTaisOne";

create table tb_usuario(
	id_usuario serial primary key,
	tx_nome varchar(200),
	nr_telefone numeric(11,0), default 55555555555 torna este um  valor padrão, ou seja enquanto não alterar este vai ser o padrão. 
	tx_senha varchar(200), 
	tx_email varchar(200),  not null, // torna o valor obrigatório 
	dt_cadastro timestamp
);

create table tb_produto(
	id_produto serial primary key,
	tx_nome varchar(200),
	vl_preco numeric (5,2),
	nr_quantidade numeric(11,0),
	vl_peso numeric (11,2)
);

create table tb_paciente(
	id_paciente serial primary key,
	tx_nome varchar(200),
	tx_especie varchar(200),
	cd_usuario integer references tb_usuario (id_usuario),
	vl_peso numeric (5,2),
	vl_idade numeric (5,2)
);
